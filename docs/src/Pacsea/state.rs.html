1//! Core application state types for Pacsea's TUI.
2//!
3//! This module defines the serializable data structures used across the
4//! application: package descriptors, search coordination types, UI focus and
5//! modals, and the central [`AppState`] container mutated by the event and UI
6//! layers. Many of these types are persisted between runs.
7use ratatui::widgets::ListState;
8use std::{collections::HashMap, path::PathBuf, time::Instant};
9
10/// Package source origin.
11///
12/// Indicates whether a package originates from the official repositories or
13/// the Arch User Repository.
14#[derive(Clone, Debug, serde::Serialize, serde::Deserialize)]
15pub enum Source {
16    /// Official repository package and its associated repository and target
17    /// architecture.
18    Official { repo: String, arch: String },
19    /// AUR package.
20    Aur,
21}
22
23/// Minimal package summary used in lists and search results.
24///
25/// This is compact enough to render in lists and panes. For a richer, detailed
26/// view, see [`PackageDetails`].
27#[derive(Clone, Debug, serde::Serialize, serde::Deserialize)]
28pub struct PackageItem {
29    /// Canonical package name.
30    pub name: String,
31    /// Version string as reported by the source.
32    pub version: String,
33    /// One-line description suitable for list display.
34    pub description: String,
35    /// Origin of the package (official repo or AUR).
36    pub source: Source,
37}
38
39/// Full set of details for a package, suitable for a dedicated information
40/// pane.
41#[derive(Clone, Debug, Default, serde::Serialize, serde::Deserialize)]
42pub struct PackageDetails {
43    /// Repository name (e.g., "extra").
44    pub repository: String,
45    /// Package name.
46    pub name: String,
47    /// Full version string.
48    pub version: String,
49    /// Long description.
50    pub description: String,
51    /// Target architecture.
52    pub architecture: String,
53    /// Upstream project URL (may be empty if unknown).
54    pub url: String,
55    /// SPDX or human-readable license identifiers.
56    pub licenses: Vec<String>,
57    /// Group memberships.
58    pub groups: Vec<String>,
59    /// Virtual provisions supplied by this package.
60    pub provides: Vec<String>,
61    /// Required dependencies.
62    pub depends: Vec<String>,
63    /// Optional dependencies with annotations.
64    pub opt_depends: Vec<String>,
65    /// Packages that require this package.
66    pub required_by: Vec<String>,
67    /// Packages for which this package is optional.
68    pub optional_for: Vec<String>,
69    /// Conflicting packages.
70    pub conflicts: Vec<String>,
71    /// Packages that this package replaces.
72    pub replaces: Vec<String>,
73    /// Download size in bytes, if available.
74    pub download_size: Option<u64>,
75    /// Installed size in bytes, if available.
76    pub install_size: Option<u64>,
77    /// Packager or maintainer name.
78    pub owner: String, // packager/maintainer
79    /// Build or packaging date (string-formatted for display).
80    pub build_date: String,
81}

83/// Search query sent to the background search worker.
84#[derive(Clone, Debug)]
85pub struct QueryInput {
86    /// Monotonic identifier used to correlate responses.
87    pub id: u64,
88    /// Raw query text entered by the user.
89    pub text: String,
90}
91/// Results corresponding to a prior [`QueryInput`].
92#[derive(Clone, Debug)]
93pub struct SearchResults {
94    /// Echoed identifier from the originating query.
95    pub id: u64,
96    /// Matching packages in rank order.
97    pub items: Vec<PackageItem>,
98}
99
100/// Modal dialog state for the UI.
101#[derive(Debug, Clone, Default)]
102pub enum Modal {
103    #[default]
104    None,
105    /// Informational alert with a non-interactive message.
106    Alert {
107        message: String,
108    },
109    /// Confirmation dialog for installing the given items.
110    ConfirmInstall {
111        items: Vec<PackageItem>,
112    },
113}
114
115/// Which UI pane currently has keyboard focus.
116#[derive(Debug, Clone, Copy, PartialEq, Eq)]
117pub enum Focus {
118    /// Center pane: search input and results.
119    Search,
120    /// Left pane: recent queries list.
121    Recent,
122    /// Right pane: pending install list.
123    Install,
124}

126/// Global application state shared by the event, networking, and UI layers.
127///
128/// This structure is mutated frequently in response to input and background
129/// updates. Certain subsets are persisted to disk to preserve user context
130/// across runs (e.g., recent searches, details cache, install list).
131#[derive(Debug)]
132pub struct AppState {
133    /// Current search input text.
134    pub input: String,
135    /// Current search results, most relevant first.
136    pub results: Vec<PackageItem>,
137    /// Index into `results` that is currently highlighted.
138    pub selected: usize,
139    /// Details for the currently highlighted result.
140    pub details: PackageDetails,
141    /// List selection state for the search results list.
142    pub list_state: ListState,
143    /// Active modal dialog, if any.
144    pub modal: Modal,
145    /// If `true`, show install steps without executing side effects.
146    pub dry_run: bool,
147    // Recent searches
148    /// Previously executed queries.
149    pub recent: Vec<String>,
150    /// List selection state for the Recent pane.
151    pub history_state: ListState,
152    /// Which pane is currently focused.
153    pub focus: Focus,
154    /// Timestamp of the last input edit, used for debouncing or throttling.
155    pub last_input_change: Instant,
156    /// Last value persisted for the input field, to avoid redundant writes.
157    pub last_saved_value: Option<String>,
158    // Persisted recent searches
159    /// Path where recent searches are persisted as JSON.
160    pub recent_path: PathBuf,
161    /// Dirty flag indicating `recent` needs to be saved.
162    pub recent_dirty: bool,
163
164    // Search coordination
165    /// Identifier of the latest query whose results are being displayed.
166    pub latest_query_id: u64,
167    /// Next query identifier to allocate.
168    pub next_query_id: u64,
169    // Details cache
170    /// Cache of details keyed by package name.
171    pub details_cache: HashMap<String, PackageDetails>,
172    /// Path where the details cache is persisted as JSON.
173    pub cache_path: PathBuf,
174    /// Dirty flag indicating `details_cache` needs to be saved.
175    pub cache_dirty: bool,
176
177    // Install list pane
178    /// Packages selected for installation.
179    pub install_list: Vec<PackageItem>,
180    /// List selection state for the Install pane.
181    pub install_state: ListState,
182    // Persisted install list
183    /// Path where the install list is persisted as JSON.
184    pub install_path: PathBuf,
185    /// Dirty flag indicating `install_list` needs to be saved.
186    pub install_dirty: bool,
187
188    // In-pane search (for Recent/Install panes)
189    /// Optional, transient find pattern used by pane-local search ("/").
190    pub pane_find: Option<String>,
191
192    // Official package index persistence
193    /// Path to the persisted official package index used for fast offline lookups.
194    pub official_index_path: PathBuf,
195
196    // Loading indicator for official index generation
197    /// Whether the application is currently generating the official index.
198    pub loading_index: bool,
199
200    // Track which packageâ€™s details the UI is focused on
201    /// Name of the package whose details are being emphasized in the UI, if any.
202    pub details_focus: Option<String>,
203
204    // Ring prefetch debounce state
205    /// Smooth scrolling accumulator for prefetch heuristics.
206    pub scroll_moves: u32,
207    /// Timestamp at which to resume ring prefetching, if paused.
208    pub ring_resume_at: Option<Instant>,
209    /// Whether a ring prefetch is needed soon.
210    pub need_ring_prefetch: bool,
211
212    // Clickable URL button rectangle (x, y, w, h) in terminal cells
213    /// Rectangle of the clickable URL button in terminal cell coordinates.
214    pub url_button_rect: Option<(u16, u16, u16, u16)>,
215}
