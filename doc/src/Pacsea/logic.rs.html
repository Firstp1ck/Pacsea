1//! Interaction logic for selection movement, search dispatching, and
2//! ring-based details prefetching.
3//!
4//! This module centralizes the non-UI behavior that reacts to user navigation
5//! and search input:
6//! - Maintains an "allowed" set of package names to throttle background
7//!   details fetching when the user scrolls quickly.
8//! - Provides ring-based prefetch to load details for neighbors around the
9//!   current selection, improving perceived responsiveness.
10//! - Sends debounced/throttled search queries with monotonically increasing ids
11//!   so the UI can ignore stale responses.
12//! - Manages the install list with simple de-duplication and cursor behavior.
13use tokio::sync::mpsc;
14
15use crate::state::{AppState, PackageItem};
16
17// Global allowed-names gating for details loading
18use std::collections::HashSet;
19use std::sync::{OnceLock, RwLock};
20/// Lazily-initialized global set of package names that are permitted to have
21/// details fetched right now.
22///
23/// The set is updated by `set_allowed_only_selected` and `set_allowed_ring`
24/// to balance responsiveness against network and CPU usage when the user is
25/// scrolling rapidly.
26fn allowed_set() -> &'static RwLock<HashSet<String>> {
27    static ALLOWED: OnceLock<RwLock<HashSet<String>>> = OnceLock::new();
28    ALLOWED.get_or_init(|| RwLock::new(HashSet::new()))
29}
30
31/// Returns whether details loading is currently allowed for the given package
32/// `name`.
33///
34/// If the lock cannot be acquired, this conservatively returns `true` to avoid
35/// blocking the UI with spurious denials.
36pub fn is_allowed(name: &str) -> bool {
37    allowed_set().read().ok().map(|s| s.contains(name)).unwrap_or(true)
38}
39
40/// Restrict details loading to only the currently selected package.
41///
42/// This is used during fast scrolling to minimize wasted work. If there is no
43/// current selection, this function is a no-op.
44pub fn set_allowed_only_selected(app: &AppState) {
45    if let Some(sel) = app.results.get(app.selected) {
46        if let Ok(mut w) = allowed_set().write() {
47            w.clear();
48            w.insert(sel.name.clone());
49        }
50    }
51}

53/// Allow details loading for a "ring" around the current selection.
54///
55/// The ring consists of the selected package plus up to `radius` neighbors
56/// above and below (clamped to list bounds). This enables background prefetch
57/// for items the user is likely to navigate to next.
58pub fn set_allowed_ring(app: &AppState, radius: usize) {
59    let mut ring: HashSet<String> = HashSet::new();
60    if let Some(sel) = app.results.get(app.selected) { ring.insert(sel.name.clone()); }
61    let len = app.results.len();
62    let mut step = 1usize;
63    while step <= radius {
64        if let Some(i) = app.selected.checked_sub(step) { if let Some(it) = app.results.get(i) { ring.insert(it.name.clone()); } }
65        let below = app.selected + step; if below < len { if let Some(it) = app.results.get(below) { ring.insert(it.name.clone()); } }
66        step += 1;
67    }
68    if let Ok(mut w) = allowed_set().write() { *w = ring; }
69}

71/// Send the current query text over the search channel with a fresh id.
72///
73/// Side effects on `app`:
74/// - Increments and records `next_query_id`
75/// - Updates `latest_query_id` to the id sent
76///
77/// The id allows the receiver to tag results so the UI can discard any stale
78/// responses that arrive out of order.
79pub fn send_query(app: &mut AppState, query_tx: &mpsc::UnboundedSender<crate::state::QueryInput>) {
80    let id = app.next_query_id;
81    app.next_query_id += 1;
82    app.latest_query_id = id;
83    let _ = query_tx.send(crate::state::QueryInput { id, text: app.input.clone() });
84}

86/// Move the selection by `delta` and update cached details and prefetch policy.
87///
88/// Behavior:
89/// - Clamps the selection to the valid range and updates the list state.
90/// - Focuses the details pane on the newly selected item and immediately shows
91///   a placeholder based on known metadata.
92/// - If cached details are present, uses them; otherwise requests loading via
93///   `details_tx`.
94/// - Tracks cumulative scroll moves to detect fast scrolling.
95/// - During fast scrolls, temporarily tightens the allowed set to only the
96///   current selection and defers ring prefetch for ~200ms.
97/// - For small scrolls, expands allowed set to a 30-item ring and begins
98///   background prefetch.
99pub fn move_sel_cached(
100    app: &mut AppState,
101    delta: isize,
102    details_tx: &mpsc::UnboundedSender<PackageItem>,
103) {
104    if app.results.is_empty() { return; }
105    let len = app.results.len() as isize;
106    let mut idx = app.selected as isize + delta;
107    if idx < 0 { idx = 0; }
108    if idx >= len { idx = len - 1; }
109    app.selected = idx as usize;
110    app.list_state.select(Some(app.selected));
111    if let Some(item) = app.results.get(app.selected).cloned() {
112        // Focus details on the currently selected item only
113        app.details_focus = Some(item.name.clone());
114
115        // Update details pane immediately with a placeholder reflecting the selection
116        app.details.name = item.name.clone();
117        app.details.version = item.version.clone();
118        app.details.description.clear();
119        match &item.source {
120            crate::state::Source::Official { repo, arch } => { app.details.repository = repo.clone(); app.details.architecture = arch.clone(); }
121            crate::state::Source::Aur => { app.details.repository = "AUR".to_string(); app.details.architecture = "any".to_string(); }
122        }
123
124        if let Some(cached) = app.details_cache.get(&item.name).cloned() { app.details = cached; } else { let _ = details_tx.send(item); }
125    }
126
127    // Debounce ring prefetch when scrolling fast (>5 items cumulatively)
128    let abs_delta_usize: usize = if delta < 0 { (-delta) as usize } else { delta as usize };
129    if abs_delta_usize > 0 {
130        let add = abs_delta_usize.min(u32::MAX as usize) as u32;
131        app.scroll_moves = app.scroll_moves.saturating_add(add);
132    }
133    if app.need_ring_prefetch {
134        // tighten allowed set to only current selection during fast scroll
135        set_allowed_only_selected(app);
136        app.ring_resume_at = Some(std::time::Instant::now() + std::time::Duration::from_millis(200));
137        return;
138    }
139    if app.scroll_moves > 5 {
140        app.need_ring_prefetch = true;
141        set_allowed_only_selected(app);
142        app.ring_resume_at = Some(std::time::Instant::now() + std::time::Duration::from_millis(200));
143        return;
144    }
145
146    // For small/slow scrolls, allow ring and prefetch immediately
147    set_allowed_ring(app, 30);
148    ring_prefetch_from_selected(app, details_tx);
149}

151/// Prefetch details for items near the current selection, alternating above and
152/// below within a fixed radius.
153///
154/// Only enqueues requests for names allowed by `is_allowed` and not already in
155/// the cache. This function is designed to be cheap and safe to call often.
156pub fn ring_prefetch_from_selected(app: &mut AppState, details_tx: &mpsc::UnboundedSender<PackageItem>) {
157    let len_u = app.results.len();
158    if len_u == 0 { return; }
159    let max_radius: usize = 30;
160    let mut step: usize = 1;
161    loop {
162        let mut progressed = false;
163        if let Some(i) = app.selected.checked_sub(step) {
164            if let Some(it) = app.results.get(i).cloned() {
165                if is_allowed(&it.name) && !app.details_cache.contains_key(&it.name) { let _ = details_tx.send(it); }
166            }
167            progressed = true;
168        }
169        let below = app.selected + step;
170        if below < len_u {
171            if let Some(it) = app.results.get(below).cloned() {
172                if is_allowed(&it.name) && !app.details_cache.contains_key(&it.name) { let _ = details_tx.send(it); }
173            }
174            progressed = true;
175        }
176        if step >= max_radius || !progressed { break; }
177        step += 1;
178    }
179}

181/// Add a `PackageItem` to the install list if it is not already present.
182///
183/// The insertion happens at the front of the list. The list is de-duplicated by
184/// case-insensitive package name. This marks the install list dirty and moves
185/// the install list cursor to the first item to keep it visible.
186pub fn add_to_install_list(app: &mut AppState, item: PackageItem) {
187    if app
188        .install_list
189        .iter()
190        .any(|p| p.name.eq_ignore_ascii_case(&item.name))
191    {
192        return;
193    }
194    app.install_list.insert(0, item);
195    app.install_dirty = true;
196    // Always keep cursor on top after adding
197    app.install_state.select(Some(0));
198}
