1//! Color palette definitions for Pacsea's TUI.
2//!
3//! This module exposes a small, opinionated theme used throughout the user
4//! interface. Colors are grouped into neutrals (base/mantle/crust/surfaces),
5//! overlays/subtexts, and accents for highlighting and semantic states.
6use ratatui::style::Color;
7
8/// Application theme palette used by rendering code.
9///
10/// All colors are provided as [`ratatui::style::Color`] and are suitable for
11/// direct use with widgets and styles.
12pub struct Theme {
13    /// Primary background color for the canvas.
14    pub base: Color,
15    /// Slightly lighter background layer used behind panels.
16    pub mantle: Color,
17    /// Darkest background shade for deep contrast areas.
18    pub crust: Color,
19    /// Subtle surface color for component backgrounds (level 1).
20    pub surface1: Color,
21    /// Subtle surface color for component backgrounds (level 2).
22    pub surface2: Color,
23    /// Muted overlay line/border color (primary).
24    pub overlay1: Color,
25    /// Muted overlay line/border color (secondary).
26    pub overlay2: Color,
27    /// Primary foreground text color.
28    pub text: Color,
29    /// Secondary text for less prominent content.
30    pub subtext0: Color,
31    /// Tertiary text for captions and low-emphasis content.
32    pub subtext1: Color,
33    /// Accent color commonly used for selection and interactive highlights.
34    pub sapphire: Color,
35    /// Accent color for emphasized headings or selections.
36    pub mauve: Color,
37    /// Success/positive state color.
38    pub green: Color,
39    /// Warning/attention state color.
40    pub yellow: Color,
41    /// Error/danger state color.
42    pub red: Color,
43    /// Accent color for subtle emphasis and borders.
44    pub lavender: Color,
45}
46
47/// Construct a [`Color::Rgb`] from an 8-bit RGB triplet.
48///
49/// This is a small helper to keep the palette definition concise.
50fn hex(rgb: (u8, u8, u8)) -> Color {
51    Color::Rgb(rgb.0, rgb.1, rgb.2)
52}
53
54/// Return the application's default theme palette.
55///
56/// Example
57///
58/// ```rust
59/// use pacsea::theme::theme;
60/// let t = theme();
61/// let primary_text = t.text;
62/// ```
63pub fn theme() -> Theme {
64    Theme {
65        base: hex((0x1e, 0x1e, 0x2e)),
66        mantle: hex((0x18, 0x18, 0x25)),
67        crust: hex((0x11, 0x11, 0x1b)),
68        surface1: hex((0x45, 0x47, 0x5a)),
69        surface2: hex((0x58, 0x5b, 0x70)),
70        overlay1: hex((0x7f, 0x84, 0x9c)),
71        overlay2: hex((0x93, 0x99, 0xb2)),
72        text: hex((0xcd, 0xd6, 0xf4)),
73        subtext0: hex((0xa6, 0xad, 0xc8)),
74        subtext1: hex((0xba, 0xc2, 0xde)),
75        sapphire: hex((0x74, 0xc7, 0xec)),
76        mauve: hex((0xcb, 0xa6, 0xf7)),
77        green: hex((0xa6, 0xe3, 0xa1)),
78        yellow: hex((0xf9, 0xe2, 0xaf)),
79        red: hex((0xf3, 0x8b, 0xa8)),
80        lavender: hex((0xb4, 0xbe, 0xfe)),
81    }
82}
